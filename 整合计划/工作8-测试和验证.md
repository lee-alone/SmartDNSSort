## 测试用例示例

### 配置加载测试
```go
func TestLoadRecursiveConfig(t *testing.T) {
    // 测试正常加载
    config, err := LoadRecursiveConfig("config/recursive.json")
    assert.NoError(t, err)
    assert.Equal(t, true, config.Enabled)
    assert.Equal(t, 5335, config.Port)
    
    // 测试文件不存在
    _, err = LoadRecursiveConfig("nonexistent.json")
    assert.Error(t, err)
}
```

### 递归查询测试
```go
func TestRecursiveQuery(t *testing.T) {
    // 测试简单查询
    result, err := resolver.Query("example.com")
    assert.NoError(t, err)
    assert.NotNil(t, result)
    
    // 测试缓存命中
    result2, err := resolver.Query("example.com")
    assert.NoError(t, err)
    assert.Equal(t, result, result2)
}
```

### 故障转移测试
```go
func TestFailover(t *testing.T) {
    // 模拟递归模块故障
    resolver.Stop()
    
    // 查询应该转移到外部 DNS
    result, err := forwarder.Query("example.com")
    assert.NoError(t, err)
    assert.NotNil(t, result)
    
    // 恢复递归模块
    resolver.Start()
    
    // 查询应该回到递归模块
    result2, err := forwarder.Query("example.com")
    assert.NoError(t, err)
    assert.NotNil(t, result2)
}
```
